  Numerals from 10\duo\ to B0\duo\ are suffixed with \conlang{-vešy}:

    \begin{longtable}[l]{r Il}
      10\duo & javešy        \\
      20\duo & vítvešy       \\
      30\duo & korvešy       \\
      40\duo & qesavešy      \\
      50\duo & pecvešy     \\
      70\duo & kušvešy      \\
      A0\duo & mežavešy     \\
      BB\duo & tuvavešy-tuva \\
    \end{longtable}

  Numerals from 100\duo\ to B00\duo\ are suffixed with \conlang{-tus}:

    \begin{longtable}[l]{r Il}
      100\duo & ertus                \\
      200\duo & víttus               \\
      300\duo & kortus               \\
      409\duo & qesetus-jukka        \\
      752\duo & kuštus-pecvešy-vít \\
    \end{longtable}
